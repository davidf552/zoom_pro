#This eliminates duplicate data
CREATE OR REPLACE TABLE `video-490706.game_sales.Videogame_sales`
(
  title STRING,
  console STRING,
  genre STRING,
  publisher STRING,
  developer STRING,
  critic_score FLOAT64,
  total_sales FLOAT64,
  release_year INT64
)
CLUSTER BY genre, console;
